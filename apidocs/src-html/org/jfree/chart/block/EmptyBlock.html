037package org.jfree.chart.block;
038
039import java.awt.Graphics2D;
040import java.awt.geom.Rectangle2D;
041import java.io.Serializable;
042import org.jfree.chart.ui.Size2D;
043import org.jfree.chart.util.PublicCloneable;
044
045/**
046 * An empty block with a fixed size.
047 */
048public class EmptyBlock extends AbstractBlock
049        implements Block, Cloneable, PublicCloneable, Serializable {
050
051    /** For serialization. */
052    private static final long serialVersionUID = -4083197869412648579L;
053
054    /**
055     * Creates a new block with the specified width and height.
056     *
057     * @param width  the width.
058     * @param height  the height.
059     */
060    public EmptyBlock(double width, double height) {
061        setWidth(width);
062        setHeight(height);
063    }
064
065    /**
066     * Arranges the contents of the block, within the given constraints, and
067     * returns the block size.
068     *
069     * @param g2  the graphics device.
070     * @param constraint  the constraint ({@code null} not permitted).
071     *
072     * @return The block size (in Java2D units, never {@code null}).
073     */
074    @Override
075    public Size2D arrange(Graphics2D g2, RectangleConstraint constraint) {
076        Size2D base = new Size2D(calculateTotalWidth(getWidth()),
077                calculateTotalHeight(getHeight()));
078        return constraint.calculateConstrainedSize(base);
079    }
080
081    /**
082     * Draws the block.  Since the block is empty, there is nothing to draw
083     * except the optional border.
084     *
085     * @param g2  the graphics device.
086     * @param area  the area.
087     */
088    @Override
089    public void draw(Graphics2D g2, Rectangle2D area) {
090        draw(g2, area, null);
091    }
092
093    /**
094     * Draws the block within the specified area.  Since the block is empty,
095     * there is nothing to draw except the optional border.
096     *
097     * @param g2  the graphics device.
098     * @param area  the area.
099     * @param params  ignored ({@code null} permitted).
100     *
101     * @return Always {@code null}.
102     */
103    @Override
104    public Object draw(Graphics2D g2, Rectangle2D area, Object params) {
105        area = trimMargin(area);
106        drawBorder(g2, area);
107        return null;
108    }
109
110    /**
111     * Returns a clone of the block.
112     *
113     * @return A clone.
114     *
115     * @throws CloneNotSupportedException if there is a problem cloning.
116     */
117    @Override
118    public Object clone() throws CloneNotSupportedException {
119        return super.clone();
120    }
121
122}
