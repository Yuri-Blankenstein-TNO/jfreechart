039package org.jfree.chart.annotations;
040
041import java.awt.Graphics2D;
042import java.awt.Shape;
043import java.awt.geom.Rectangle2D;
044import java.util.Objects;
045
046import org.jfree.chart.axis.ValueAxis;
047import org.jfree.chart.entity.EntityCollection;
048import org.jfree.chart.entity.XYAnnotationEntity;
049import org.jfree.chart.plot.PlotRenderingInfo;
050import org.jfree.chart.plot.XYPlot;
051
052/**
053 * The interface that must be supported by annotations that are to be added to
054 * an {@link XYPlot}.
055 */
056public abstract class AbstractXYAnnotation extends AbstractAnnotation
057        implements XYAnnotation {
058
059    /** The tool tip text. */
060    private String toolTipText;
061
062    /** The URL. */
063    private String url;
064
065    /**
066     * Creates a new instance that has no tool tip or URL specified.
067     */
068    protected AbstractXYAnnotation() {
069        super();
070        this.toolTipText = null;
071        this.url = null;
072    }
073
074    /**
075     * Returns the tool tip text for the annotation.  This will be displayed in
076     * a {@link org.jfree.chart.ChartPanel} when the mouse pointer hovers over
077     * the annotation.
078     *
079     * @return The tool tip text (possibly {@code null}).
080     *
081     * @see #setToolTipText(String)
082     */
083    public String getToolTipText() {
084        return this.toolTipText;
085    }
086
087    /**
088     * Sets the tool tip text for the annotation.
089     *
090     * @param text  the tool tip text ({@code null} permitted).
091     *
092     * @see #getToolTipText()
093     */
094    public void setToolTipText(String text) {
095        this.toolTipText = text;
096    }
097
098    /**
099     * Returns the URL for the annotation.  This URL will be used to provide
100     * hyperlinks when an HTML image map is created for the chart.
101     *
102     * @return The URL (possibly {@code null}).
103     *
104     * @see #setURL(String)
105     */
106    public String getURL() {
107        return this.url;
108    }
109
110    /**
111     * Sets the URL for the annotation.
112     *
113     * @param url  the URL ({@code null} permitted).
114     *
115     * @see #getURL()
116     */
117    public void setURL(String url) {
118        this.url = url;
119    }
120
121    /**
122     * Draws the annotation.
123     *
124     * @param g2  the graphics device.
125     * @param plot  the plot.
126     * @param dataArea  the data area.
127     * @param domainAxis  the domain axis.
128     * @param rangeAxis  the range axis.
129     * @param rendererIndex  the renderer index.
130     * @param info  if supplied, this info object will be populated with
131     *              entity information.
132     */
133    @Override
134    public abstract void draw(Graphics2D g2, XYPlot plot, Rectangle2D dataArea,
135                              ValueAxis domainAxis, ValueAxis rangeAxis,
136                              int rendererIndex,
137                              PlotRenderingInfo info);
138
139    /**
140     * A utility method for adding an {@link XYAnnotationEntity} to
141     * a {@link PlotRenderingInfo} instance.
142     *
143     * @param info  the plot rendering info ({@code null} permitted).
144     * @param hotspot  the hotspot area.
145     * @param rendererIndex  the renderer index.
146     * @param toolTipText  the tool tip text.
147     * @param urlText  the URL text.
148     * @see #createEntity(Shape, int, String, String)
149     */
150    protected void addEntity(PlotRenderingInfo info,
151                             Shape hotspot, int rendererIndex,
152                             String toolTipText, String urlText) {
153        if (info == null) {
154            return;
155        }
156        EntityCollection entities = info.getOwner().getEntityCollection();
157        if (entities == null) {
158            return;
159        }
160        XYAnnotationEntity entity = createEntity(hotspot, rendererIndex,
161                toolTipText, urlText);
162        entities.add(entity);
163    }
164    
165    /**
166     * A factory method for creating an {@link XYAnnotationEntity} for this
167     * annotation.
168     * 
169     * @param hotspot       the hotspot area.
170     * @param rendererIndex the renderer index.
171     * @param toolTipText   the tool tip text.
172     * @param urlText       the URL text.
173     * @return the created entity
174     */
175    protected XYAnnotationEntity createEntity(Shape hotspot, int rendererIndex,
176            String toolTipText, String urlText) {
177        return new XYAnnotationEntity(hotspot, this, rendererIndex, toolTipText,
178                urlText);
179    }
180
181    /**
182     * Tests this annotation for equality with an arbitrary object.
183     *
184     * @param obj  the object ({@code null} permitted).
185     *
186     * @return A boolean.
187     */
188    @Override
189    public boolean equals(Object obj) {
190        if (obj == this) {
191            return true;
192        }
193        if (!(obj instanceof AbstractXYAnnotation)) {
194            return false;
195        }
196        AbstractXYAnnotation that = (AbstractXYAnnotation) obj;
197        if (!Objects.equals(this.toolTipText, that.toolTipText)) {
198            return false;
199        }
200        if (!Objects.equals(this.url, that.url)) {
201            return false;
202        }
203
204        // fix the "equals not symmetric" problem
205        if (!that.canEqual(this)) {
206            return false;
207        }
208
209        return super.equals(obj);
210    }
211
212    /**
213     * Ensures symmetry between super/subclass implementations of equals. For
214     * more detail, see http://jqno.nl/equalsverifier/manual/inheritance.
215     *
216     * @param other Object
217     * 
218     * @return true ONLY if the parameter is THIS class type
219     */
220    @Override
221    public boolean canEqual(Object other) {
222        // fix the "equals not symmetric" problem
223        return (other instanceof AbstractXYAnnotation);
224    }
225
226    /**
227     * Returns a hash code for this instance.
228     *
229     * @return A hash code.
230     */
231    @Override
232    public int hashCode() {
233        int result = super.hashCode(); // equals calls superclass, hashCode must also
234        result = 37 * result + Objects.hashCode(this.toolTipText);
235        result = 37 * result + Objects.hashCode(this.url);
236        return result;
237    }
238
239}
