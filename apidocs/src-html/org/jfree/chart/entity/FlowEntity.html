037package org.jfree.chart.entity;
038
039import java.awt.Shape;
040import java.util.Objects;
041import org.jfree.chart.plot.flow.FlowPlot;
042import org.jfree.chart.util.Args;
043import org.jfree.data.flow.FlowKey;
044
045/**
046 * A chart entity representing the flow between two nodes in a {@link FlowPlot}.
047 * 
048 * @since 1.5.3
049 */
050public class FlowEntity extends ChartEntity {
051
052    private FlowKey key;
053    
054    /**
055     * Creates a new instance.
056     * 
057     * @param key  the key identifying the flow ({@code null} not permitted).
058     * @param area  the outline of the entity ({@code null} not permitted).
059     * @param toolTipText  the tool tip text.
060     * @param urlText  the URL text.
061     */
062    public FlowEntity(FlowKey key, Shape area, String toolTipText, String urlText) {
063        super(area, toolTipText, urlText);
064        Args.nullNotPermitted(key, "key");
065        this.key = key;
066    }
067    
068    /**
069     * Returns the key identifying the flow.
070     * 
071     * @return The flow key (never {@code null}). 
072     */
073    public FlowKey getKey() {
074        return this.key;
075    }
076
077    /**
078     * Returns a string representation of this instance, primarily for 
079     * debugging purposes.
080     * 
081     * @return A string. 
082     */
083    @Override
084    public String toString() {
085        return "[FlowEntity: " + this.key + "]";
086    }
087
088    /**
089     * Tests this instance for equality with an arbitrary object.
090     * 
091     * @param obj  the object ({@code null} permitted).
092     * 
093     * @return A boolean. 
094     */
095    @Override
096    public boolean equals(Object obj) {
097        if (!(obj instanceof FlowEntity)) {
098            return false;
099        }
100        FlowEntity that = (FlowEntity) obj;
101        if (!java.util.Objects.equals(this.key, that.key)) {
102            return false;
103        }
104        // fix the "equals not symmetric" problem
105        if (!that.canEqual(this)) {
106            return false;
107        }
108
109        return super.equals(obj);
110    }
111
112    /**
113     * Ensures symmetry between super/subclass implementations of equals. For
114     * more detail, see http://jqno.nl/equalsverifier/manual/inheritance.
115     *
116     * @param other Object
117     * 
118     * @return true ONLY if the parameter is THIS class type
119     */
120    @Override
121    public boolean canEqual(Object other) {
122        // fix the "equals not symmetric" problem
123        return (other instanceof FlowEntity);
124    }
125
126    @Override
127    public int hashCode() {
128        int hash = super.hashCode();
129        hash = 79 * hash + Objects.hashCode(this.key);
130        return hash;
131    }
132
133}
