037package org.jfree.data.function;
038
039import java.io.Serializable;
040
041import org.jfree.chart.HashUtils;
042
043/**
044 * A function in the form y = a + bx.
045 */
046public class LineFunction2D implements Function2D, Serializable {
047
048    /** The intercept. */
049    private double a;
050
051    /** The slope of the line. */
052    private double b;
053
054    /**
055     * Constructs a new line function.
056     *
057     * @param a  the intercept.
058     * @param b  the slope.
059     */
060    public LineFunction2D(double a, double b) {
061        this.a = a;
062        this.b = b;
063    }
064
065    /**
066     * Returns the 'a' coefficient that was specified in the constructor.
067     *
068     * @return The 'a' coefficient.
069     */
070    public double getIntercept() {
071        return this.a;
072    }
073
074    /**
075     * Returns the 'b' coefficient that was specified in the constructor.
076     *
077     * @return The 'b' coefficient.
078     */
079    public double getSlope() {
080        return this.b;
081    }
082
083    /**
084     * Returns the function value.
085     *
086     * @param x  the x-value.
087     *
088     * @return The value.
089     */
090    @Override
091    public double getValue(double x) {
092        return this.a + this.b * x;
093    }
094
095    /**
096     * Tests this function for equality with an arbitrary object.
097     *
098     * @param obj  the object ({@code null} permitted).
099     *
100     * @return A boolean.
101     */
102    @Override
103    public boolean equals(Object obj) {
104        if (!(obj instanceof LineFunction2D)) {
105            return false;
106        }
107        LineFunction2D that = (LineFunction2D) obj;
108        if (this.a != that.a) {
109            return false;
110        }
111        if (this.b != that.b) {
112            return false;
113        }
114        return true;
115    }
116
117    /**
118     * Returns a hash code for this instance.
119     * 
120     * @return A hash code.
121     */
122    @Override
123    public int hashCode() {
124        int result = 29;
125        result = HashUtils.hashCode(result, this.a);
126        result = HashUtils.hashCode(result, this.b);
127        return result;
128    }
129}
